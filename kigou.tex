\chapter*{記号一覧}
\addcontentsline{toc}{chapter}{記号一覧}
\begin{itemize}
\setlength{\itemindent}{-1zw}
\setlength{\leftskip}{-2zw}
\setlength{\labelsep}{0.5zw}
\item[]$\ZZ$：整数全体の集合（整数環）$\Set{0, \pm 1, \pm 2, \dots}$。
\item[]$\RR$：実数全体の集合（実数体）。
\item[]$\CC$：複素数全体の集合（複素数体）。
\item[]$|S|$：集合$S$に対してその元（要素）の個数。
\item[]$\sum_{i=a}^b x_i$：$\Set{x_a, x_{a+1}, \dots, x_b}$の和。
\item[]$\prod_{i=a}^b x_i$：$\Set{x_a, x_{a+1}, \dots, x_b}$の積。
\item[]$a\oplus b$：$a$と$b$の排他的論理和。
\item[]$O(f(n))$：ランダウの$O$記法。$f(n)$が$O(g(n))$なのは、ある定数$c$があって十分大きい全ての$n$について$|f(n)/g(n)|<c$。
\item[]$a:=b$：$b$で$a$を定義する。
\item[]$a \equiv b \bmod{m}$：$a-b$が$m$で割り切れる。
\item[]$a \pmod{m}$：整数$q$, $r$に対して$a=qm+r$（$0 \le r < m$）となる$r$。
\item[]$\ZZ/m \ZZ$：整数を$m$で割った余りの集合（剰余環）$\Set{0, 1, \dots, m-1}$。
\item[]$\FF_p$：$p$個の元$\Set{0, 1, \dots, p-1}$からなる有限体。
\item[]$\Char(k)$：体$k$の標数。
\item[]${\FF_p}^*$：0を除いた$\FF_p$の元全体。
\item[]$a||b$：二つのデータ$a$と$b$を連結したもの。
\item[]$a \eqq b$：$a$と$b$が等しいかを確認する。
\item[]$\gen{g}$：$g$で生成される巡回群$\Set{g^0, g^1, \dots}$。
\item[]$a \cdot b$：2個のベクトル$a=(a_1,\dots,a_n)$, $b=(b_1,\dots,b_n)$の内積$\sum_{i=1}^n a_i b_i$。
\item[]$R[x\textrm{]}$：変数$x$についての$R$係数多項式全体
$\Set{\sum_{i=0}^n r_i x^i|r_i \in R, n = 0, 1, \dots}$。
\item[]文末の$\Box$：証明終わり。
\item[]$R[x\textrm{]}/(f(x))$：$R$係数多項式を$f(x)$で割った余りの多項式全体。
\item[]$\gcd(a,b)$：$a$と$b$の最大公約数。
\item[]$\combi{n}{k}$：$n$個から$k$個取り出す組み合わせの数。
\item[]$\FF_{p^n}$：有限体$\FF_p$の$n$次拡大体。
\item[]$\overline{k}$：体$k$の代数的閉包。
\item[]$\ord{a}{f}$：関数$f(x)$の点$x=a$における位数。
\item[]$E/k$：体$k$上の楕円曲線$\Set{(x,y)\in \overline{k}^2|y^2=x^3+ax+b}\cup \Set{O}$。\\
ただし$a$, $b \in k$について$4a^3+27b^2 \ne 0$。
\item[]$E(h)$：体$h$に対して$h$有理点の集合$E(h):=(E \cap h^2)\cup \Set{O}$。
\item[]$\deg{D}$：因子$D$の次数。
\item[]$\Div(f)$：関数$f$の主因子。
\item[]$E[m\textrm{]}$：楕円曲線$E$の$m$等分点（ねじれ点）の集合$\Set{P \in E|mP = O}$。
\item[]$e$：体$k$上の楕円曲線$E$のペアリング$e:E[m] \times E[m] \rightarrow \overline{k}$。
\end{itemize}
